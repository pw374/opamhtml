Html tags generators

val make_info : Cow.Html.t option -> Cow.Html.t
Generate a <div class="info">xxx</div> tag or Html.nil if None
val make_pre : Cow.Html.t -> Cow.Html.t
Wrap the content into a <pre> html tag
val make_span : ?css_class:string -> Cow.Html.t -> Cow.Html.t
Wrap the content into a <span> html tag with a class label option
val keyword : string -> Cow.Html.t
Wrap the content into a <span class="keyword"> html tag
val constructor : string -> Cow.Html.t
Wrap the content into a <span class="constructor"> html tag
val code : string -> Cow.Html.t -> Cow.Html.t
Wrap the content into a \<code\> html tag with a class label option
val make_type_table : Cow.Html.t list -> Cow.Html.t
Wrap the column list into a <table class="typetable"> by wraping the elements with <tr> tags. The elements should already have the appropriates <td> tags
val generate_mark : Opam_doc_config.mark -> string -> Cow.Html.t -> Cow.Html.t
Wrap the content into a <span> that will contain a class attribute with the unique id used by the javascript to lookup types, classes, values, ... The html inlining prevents using id because two tags with the attribute : id="TYPEt" could also be defined in submodules and therefore false the search. Html_utils.generate_mark mark element_name content
val make_field_comment : Cow.Html.t -> Cow.Html.t
Generate the <td> for a class field comment
val make_variant_cell : string -> string -> Cow.Html.t list -> Cow.Html.t option -> Cow.Html.t
Generate the <td> for a variant element
val make_record_label_cell : string -> string -> bool -> Cow.Html.t -> Cow.Html.t option -> Cow.Html.t
Generate the <td> for a record element
val html_of_type_param_list : Cow.Html.t list -> [< `Negative | `None | `Positive ] list -> Cow.Html.t
Generate the type parameters for a type signature. e.g: (+'a,-'b) for the type (+'a,-'b) t
val html_of_type_class_param_list : Cow.Html.t list -> [< `Negative | `None | `Positive ] list -> Cow.Html.t
Generate the class parameters for a class signature. e.g: (+'a,-'b) for the type (+'a,-'b) t
val js_array_of_include_items : Types.module_type -> string
Parse the include's module_type to retrieve the contained elements and return it as a json array
